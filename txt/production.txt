    SELECT COUNT(*) as count 
    FROM batches 
    WHERE DATE(scheduled_at) = CURDATE() 
      AND is_deleted = 0
");

    SELECT COUNT(*) as count

    WHERE status = 'in_progress' AND is_deleted = 0
");

    SELECT COUNT(*) as count 
    FROM batches

");
